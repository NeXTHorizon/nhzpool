CREATE TABLE blocks (
timestamp NUMERIC,
block TEXT,
totalfee NUMERIC,
UNIQUE (timestamp));

CREATE TABLE accounts (
blocktime NUMERIC,
account TEXT,
percentage NUMERIC,
amount NUMERIC,
paid NUMERIC,
UNIQUE (blocktime, account));

CREATE TABLE leased (
account TEXT,
heightfrom NUMERIC,
heightto NUMERIC,
amount NUMERIC,
UNIQUE (account));

CREATE TABLE payouts (
account TEXT,
fee NUMERIC,
payment NUMERIC,
UNIQUE (account, payment))
